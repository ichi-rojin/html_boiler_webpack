Window information

/src/public/assets/scss/document/ui/_windowInfo.scss
$(function () {
  const resizeHandler = () => {
    const vw = document.documentElement.style.getPropertyValue('--vw');
    const vh = document.documentElement.style.getPropertyValue('--vh');
    const sw = document.documentElement.style.getPropertyValue('--sw');

    $('.windowWidth:first').text("Window width: " + vw);
    $('.windowHeight:first').text("Window height: " + vh);
    $('.scrollWidth:first').text("Scrollbar width: " + sw);
  };
  setTimeout(resizeHandler, 300);
  $(window).on('resize', () => {
    setTimeout(resizeHandler, 300)
  });
});
